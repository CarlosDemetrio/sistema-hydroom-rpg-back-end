 /**

  *
  * @param jogoId        ID do jogo
  * @param nome          Nome da categoria

  * @param cor           Cor em formato #RRGGBB
  * @param ordemExibicao Ordem de exibição
  */

     String descricao,
 
     @Size(max = 7, message = "Cor deve estar no formato #RRGGBB")
     String cor,
 
     Integer ordemExibicao
 ) {}
